'use strict';

let BaseCard = require('./BaseCard');

/**
 * 标准卡片
 *
 * @class
 * @extends {BaseCard}
 *
 * @example
 * let standardCard = new StandardCard();
 * standardCard.setTitle('title');
 * standardCard.setContent('content');
 * standardCard.setImage('http://www.image');
 */
class StandardCard extends BaseCard {
    constructor () {
        super(['title', 'content', 'image']);
        this.data.type = 'standard';
    }
}

module.exports = StandardCard;
